168

224  solver.compute();
225  auto eigenValues(solver.eigenvalues());

227 
228  // Use std::sort to sort an indices vector in the same way eigen values should be ordered
229  // initialize original index locations
230  std::vector<size_t> idx(eigenValues.size());
231  std::iota(idx.begin(), idx.end(), 0);
232  // sort indexes based on comparing values in eigenValues
233  std::sort(idx.begin(), idx.end(),
234  [&eigenValues](size_t i1, size_t i2) {return eigenValues[i1] > eigenValues[i2];});
235 
236  // Build D the diagonal matrix with ordered eigen values in diagonal
237  auto D(Matrix<Scalar>::Zero(latentDim, latentDim));
238  # pragma omp parallel for
239  for (i=0; i<latentDim; ++i)
240  {
241  // use sorted indices to retrieve eigen values in descending order
242  D(i,i) = std::sqrt(eigenValues[idx[i]]);
243  }
244 
245  // Build V the transposed matrix of ordered eigen vectors
246  Matrix<Scalar> V(latentDim, eigenVectors.cols());
247  # pragma omp parallel for
248  for (i=0; i<latentDim; ++i)
249  {
250  // use sorted indices to retrieve eigen vectors in descending order
251  V.row(i) = eigenVectors.col(idx[i]);
252  }
253
