228  void OptimisationSolver<Scalar>::shermanMorissonUpdate (
229  const Matrix<Scalar>& old_K_minus1,
230  Matrix<Scalar>& new_K_minus1,
231  const Scalar& old_detK,
232  Scalar& new_detK,
233  unsigned int lv_num,
234  Vector<Scalar>& diff_cov_vector) const
235  {
236  #ifdef DEBUG
237  assert(old_detK != 0.0);
238  #endif
239 
240  Scalar centerCoeff(diff_cov_vector[lv_num]);
241 
242  // ===== One row modification =====
243  Scalar dotp1(diff_cov_vector.dot(old_K_minus1.col(lv_num)) + Scalar(1));
244  // Determinant update
245  new_detK = dotp1 * old_detK;
246 
247  // Inverse update
248  new_K_minus1.noalias() =
249  (old_K_minus1 - ( ((old_K_minus1.col(lv_num)*(diff_cov_vector.transpose()/dotp1))*old_K_minus1))
250  ).eval();
251 
252  // ===== One column modification =====
253  diff_cov_vector[lv_num] = Scalar(0);
254 
255  dotp1 = new_K_minus1.row(lv_num).dot(diff_cov_vector) + Scalar(1);
256 
257  // Determinant update
258  new_detK = dotp1 * new_detK;
259 
260  // Inverse update
261  new_K_minus1 =
262  (new_K_minus1 - ( new_K_minus1*(diff_cov_vector/dotp1)*new_K_minus1.row(lv_num))
263  ).eval();

292  {
293  // Use EigenSolver to compute eigen vector/values decomposition
294  Eigen::EigenSolver<Matrix<Scalar>> solver(ZZt, true);
295 
296  const Vector<Scalar>& eigenValues(solver.eigenvalues().real());
297  const Matrix<Scalar>& eigenVectors(solver.eigenvectors().real());
298 
299  // Use std::sort to sort an indices vector in the same way eigen values should be ordered
300  // initialize original index locations
301  std::vector<size_t> idx(eigenValues.size());
302  std::iota(idx.begin(), idx.end(), 0);
303  // sort indexes based on comparing values in eigenValues
304  std::function<size_t(size_t, size_t)> cmpIndices(
305  [&eigenValues](size_t i1, size_t i2) {
306  return eigenValues[i1] > eigenValues[i2];
307  }
308  );
309 
310  std::sort(idx.begin(), idx.end(),cmpIndices);
311 
312  // Build V the transposed matrix of ordered eigen vectors
313  Matrix<Scalar> V(latentDim, eigenVectors.cols());
314  # pragma omp parallel for
315  for (unsigned int i=0; i<latentDim; ++i)
316  {
317  // use sorted indices to retrieve eigen vectors in descending order
318  V.row(i).noalias() = eigenVectors.col(idx[i]).transpose();
319  }
320 
321  // X as column vector
322  X = Eigen::Map<Vector<Scalar>>(V.data(), latentDim*nb_data, 1);
323 
324  // normalize X
325  X = X / (std::numeric_limits<Scalar>::epsilon() + std::max(abs(X.maxCoeff()), abs(X.minCoeff())));
